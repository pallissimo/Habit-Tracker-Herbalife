  <script>

    const tg = window.Telegram?.WebApp || null;


    if (tg) {

      tg.ready();

      tg.expand();


      const bg = tg.themeParams.bg_color;

      const text = tg.themeParams.text_color;

      const button = tg.themeParams.button_color;

      const buttonText = tg.themeParams.button_text_color;

      const secondary = tg.themeParams.secondary_bg_color;


      if (bg) document.documentElement.style.setProperty('--tg-theme-bg-color', bg);

      if (text) document.documentElement.style.setProperty('--tg-theme-text-color', text);

      if (button) document.documentElement.style.setProperty('--tg-theme-button-color', button);

      if (buttonText) document.documentElement.style.setProperty('--tg-theme-button-text-color', buttonText);

      if (secondary) document.documentElement.style.setProperty('--tg-theme-secondary-bg-color', secondary);

    }


    const habits = [

      { id: "H1", text: "🥤 เชคมื้อเช้า" },

      { id: "H2", text: "🏃 ออกกำลังกาย" },

      { id: "H3", text: "💧 ดื่มน้ำเพียงพอ" },

      { id: "H4", text: "🥤 เชคแก้วที่ 2" },

      { id: "H5", text: "🍚 มื้ออาหารสุขภาพ" },

      { id: "H6", text: "🌙 นอนหลับดี" },

      { id: "H7", text: "🍔 ไม่กินของอ้วน" },

      { id: "H8", text: "🚭 ไม่สูบบุหรี่" },

      { id: "H9", text: "🍺 ไม่ดื่มแอลกอฮอล์" }

    ];


    const urlParams = new URLSearchParams(window.location.search);

    const day = urlParams.get('day') || '?';

    const type = urlParams.get('type') || 'Habit';

    const preset = (urlParams.get('selected') || '').split(',').filter(Boolean);


    const displayDay = document.getElementById('display-day');

    const container = document.getElementById('habit-container');

    const progressFill = document.getElementById('progress-fill');

    const progressText = document.getElementById('progress-text');

    const progressSubtext = document.getElementById('progress-subtext');

    const saveBtn = document.getElementById('save-btn');

    const clearBtn = document.getElementById('clear-btn');

    const helperText = document.getElementById('helper-text');

    const statusEl = document.getElementById('status');


    displayDay.innerText = `Day ${day} / ${type}`;


    function setStatus(message = '', mode = '') {

      statusEl.className = 'status';

      if (!message) {

        statusEl.textContent = '';

        return;

      }

      statusEl.textContent = message;

      if (mode) statusEl.classList.add(mode);

    }


    function getSelectedHabits() {

      return Array.from(document.querySelectorAll('input[type="checkbox"]:checked')).map(cb => cb.id);

    }


    function updateProgress() {

      const selected = getSelectedHabits();

      const total = habits.length;

      const done = selected.length;

      const percent = total === 0 ? 0 : Math.round((done / total) * 100);


      progressFill.style.width = percent + '%';

      progressText.textContent = `${done}/${total}`;


      if (done === 0) {

        progressSubtext.textContent = 'ยังไม่ได้เลือกกิจกรรม';

      } else if (done === total) {

        progressSubtext.textContent = 'ยอดเยี่ยม ทำครบทุกข้อแล้ว 🎉';

      } else {

        progressSubtext.textContent = `สำเร็จแล้ว ${percent}%`;

      }


      document.querySelectorAll('.habit-item').forEach(item => {

        const cb = item.querySelector('input[type="checkbox"]');

        item.classList.toggle('checked', cb.checked);

      });

    }


    function renderHabits() {

      container.innerHTML = '';


      habits.forEach(habit => {

        const row = document.createElement('div');

        row.className = 'habit-item';


        const checked = preset.includes(habit.id);


        row.innerHTML = `

          <input type="checkbox" id="${habit.id}" value="${habit.id}" ${checked ? 'checked' : ''}>

          <label class="habit-text" for="${habit.id}">${habit.text}</label>

        `;


        row.addEventListener('click', (e) => {

          const checkbox = row.querySelector('input[type="checkbox"]');

          if (e.target.tagName !== 'INPUT') {

            checkbox.checked = !checkbox.checked;

          }

          updateProgress();

          setStatus('');

        });


        container.appendChild(row);

      });


      updateProgress();

    }


    function buildPayload() {

      return {

        action: "SUBMIT_HABIT",

        data: getSelectedHabits(),

        day: day,

        type: type,

        total: habits.length,

        completed: getSelectedHabits().length,

        percent: Math.round((getSelectedHabits().length / habits.length) * 100)

      };

    }


    function saveData() {

      const selected = getSelectedHabits();


      if (selected.length === 0) {

        if (tg) {

          tg.showAlert("กรุณาเลือกอย่างน้อย 1 รายการครับ");

        } else {

          setStatus("กรุณาเลือกอย่างน้อย 1 รายการครับ", "error");

        }

        return;

      }


      const payload = buildPayload();


      if (tg) {

        tg.sendData(JSON.stringify(payload));

        setStatus("ส่งข้อมูลกลับไปที่ Telegram แล้ว", "ok");

      } else {

        setStatus("โหมดทดสอบ: หน้านี้ไม่ได้เปิดผ่าน Telegram", "error");

        console.log("TEST PAYLOAD:", payload);

        alert("โหมดทดสอบ: ยังไม่ได้เปิดผ่าน Telegram\n\nดูข้อมูลที่ Console ได้");

      }

    }


    function clearAll() {

      document.querySelectorAll('input[type="checkbox"]').forEach(cb => {

        cb.checked = false;

      });

      updateProgress();

      setStatus('');

    }


    renderHabits();


    saveBtn.addEventListener('click', saveData);

    clearBtn.addEventListener('click', clearAll);


    if (tg) {

      helperText.textContent = "เชื่อมต่อกับ Telegram Web App แล้ว";

      tg.MainButton.setText("บันทึกคะแนน");

      tg.MainButton.show();

      tg.MainButton.onClick(saveData);

    } else {

      helperText.textContent = "โหมดทดสอบบนเว็บ: ปุ่มบันทึกจะยังไม่ส่งกลับ Telegram";

    }
