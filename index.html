<body>


    <div class="header">

        <h2>📝 บันทึกความสำเร็จ</h2>

        <div id="display-day" class="day-info">กำลังโหลด...</div>

    </div>


    <div class="habit-list" id="habit-container">

        <!-- รายการ Habit จะถูกสร้างจาก JS -->

    </div>


    <script>

        const tg = window.Telegram.WebApp;

        tg.expand(); // ขยายหน้าจอให้เต็ม


        // รายการ Habit จากโค้ดเดิมของคุณ

        const habits = [

            { id: "H1", text: "🥤 เชคมื้อเช้า" },

            { id: "H2", text: "🏃 ออกกำลังกาย" },

            { id: "H3", text: "💧 ดื่มน้ำเพียงพอ" },

            { id: "H4", text: "🥤 เชคแก้วที่ 2" },

            { id: "H5", text: "🍚 มื้ออาหารสุขภาพ" },

            { id: "H6", text: "🌙 นอนหลับดี" },

            { id: "H7", text: "🍔 ไม่กินของอ้วน" },

            { id: "H8", text: "🚭 ไม่สูบบุหรี่" },

            { id: "H9", text: "🚭 ไม่ดื่มแอลกอฮอล์" }

        ];


        // ดึงค่า Day จาก URL (ที่ส่งมาจาก Bot)

        const urlParams = new URLSearchParams(window.location.search);

        const day = urlParams.get('day') || '?';

        const type = urlParams.get('type') || '?';

        document.getElementById('display-day').innerText = `Day ${day} / ${type}`;


        // สร้างรายการในหน้าจอ

        const container = document.getElementById('habit-container');

        habits.forEach(habit => {

            const div = document.createElement('div');

            div.className = 'habit-item';

            div.innerHTML = `

                <input type="checkbox" id="${habit.id}" value="${habit.id}">

                <label class="habit-text" for="${habit.id}">${habit.text}</label>

            `;

            // ทำให้คลิกที่แถวแล้วติ๊ก Checkbox ได้เลย

            div.onclick = (e) => {

                if(e.target.tagName !== 'INPUT') {

                    const cb = div.querySelector('input');

                    cb.checked = !cb.checked;

                }

            };

            container.appendChild(div);

        });


        // ตั้งค่าปุ่มหลัก (Main Button) ของ Telegram

        tg.MainButton.setText("บันทึกคะแนน").show();

        tg.MainButton.onClick(() => {

            const selected = [];

            document.querySelectorAll('input[type="checkbox"]:checked').forEach(cb => {

                selected.push(cb.id);

            });


            if (selected.length === 0) {

                tg.showAlert("กรุณาเลือกอย่างน้อย 1 รายการครับ");

                return;

            }


            // ส่งข้อมูลกลับไปที่ Bot

            tg.sendData(JSON.stringify({

                action: "SUBMIT_HABIT",

                data: selected,

                day: day

            }));

        });

    </script>
